-- 1. Tabela de fornecedores
CREATE TABLE IF NOT EXISTS fornecedores (
  id              BIGSERIAL PRIMARY KEY,
  nome            TEXT NOT NULL,
  cnpj            TEXT,
  categoria       TEXT,
  slug            TEXT UNIQUE NOT NULL,
  contato         TEXT,
  email           TEXT,
  status          TEXT DEFAULT 'ativo',
  ultimo_upload   TIMESTAMPTZ,
  total_pedidos   INTEGER DEFAULT 0,
  total_linhas    INTEGER DEFAULT 0,
  total_valor     NUMERIC(15,2) DEFAULT 0,
  total_kg        NUMERIC(15,3) DEFAULT 0,
  pct_sellin      NUMERIC(12,4) DEFAULT 0,
  criado_em       TIMESTAMPTZ DEFAULT now(),
  atualizado_em   TIMESTAMPTZ DEFAULT now()
);

-- 2. Tabela de pedidos por fornecedor
CREATE TABLE IF NOT EXISTS pedidos_fornecedor (
  id              BIGSERIAL PRIMARY KEY,
  fornecedor_id   BIGINT REFERENCES fornecedores(id) ON DELETE CASCADE,
  id_pedido_venda TEXT,
  vendedor        TEXT,
  tipo            TEXT,
  id_cliente      TEXT,
  cliente         TEXT,
  bairro          TEXT,
  cidade          TEXT,
  uf              TEXT,
  zona            TEXT,
  segmento        TEXT,
  data            DATE,
  dia             INTEGER,
  mes             INTEGER,
  ano             INTEGER,
  dia_semana      TEXT,
  id_produto      TEXT,
  grupo           TEXT,
  sub_grupo       TEXT,
  produto         TEXT,
  qtde_pc         NUMERIC(12,3) DEFAULT 0,
  qtde_kg         NUMERIC(12,3) DEFAULT 0,
  valor           NUMERIC(12,2) DEFAULT 0,
  margem          NUMERIC(12,4) DEFAULT 0,
  digitador       TEXT,
  validade        TEXT,
  margem_grupo    NUMERIC(12,4) DEFAULT 0,
  regiao          TEXT,
  criado_em       TIMESTAMPTZ DEFAULT now()
);

-- 3. Tabela de notas fiscais (NFe Sell-In)
-- CORRIGIDO: tabela estava faltando no SQL original
CREATE TABLE IF NOT EXISTS notas_fiscais (
  id                BIGSERIAL PRIMARY KEY,
  fornecedor_id     BIGINT REFERENCES fornecedores(id) ON DELETE CASCADE,
  numero            TEXT,
  emissao           DATE,
  cnpj_emitente     TEXT,
  emitente          TEXT,
  natureza_operacao TEXT,
  situacao          TEXT,
  valor             NUMERIC(15,2) DEFAULT 0,
  chave             TEXT,
  criado_em         TIMESTAMPTZ DEFAULT now()
);

-- 4. Índices para performance
CREATE INDEX IF NOT EXISTS idx_pedidos_fornecedor_id ON pedidos_fornecedor(fornecedor_id);
CREATE INDEX IF NOT EXISTS idx_pedidos_data ON pedidos_fornecedor(data);
CREATE INDEX IF NOT EXISTS idx_pedidos_bairro ON pedidos_fornecedor(bairro);
CREATE INDEX IF NOT EXISTS idx_pedidos_produto ON pedidos_fornecedor(produto);
CREATE INDEX IF NOT EXISTS idx_nfe_fornecedor_id ON notas_fiscais(fornecedor_id);

-- 5. Row Level Security (RLS) — leitura pública por slug
ALTER TABLE fornecedores ENABLE ROW LEVEL SECURITY;
ALTER TABLE pedidos_fornecedor ENABLE ROW LEVEL SECURITY;
ALTER TABLE notas_fiscais ENABLE ROW LEVEL SECURITY;

CREATE POLICY "leitura_publica_fornecedores" ON fornecedores FOR SELECT USING (true);
CREATE POLICY "leitura_publica_pedidos" ON pedidos_fornecedor FOR SELECT USING (true);
CREATE POLICY "leitura_publica_nfe" ON notas_fiscais FOR SELECT USING (true);
CREATE POLICY "insert_service_role_fornecedores" ON fornecedores FOR INSERT WITH CHECK (true);
CREATE POLICY "update_service_role_fornecedores" ON fornecedores FOR UPDATE USING (true);
CREATE POLICY "insert_service_role_pedidos" ON pedidos_fornecedor FOR INSERT WITH CHECK (true);
CREATE POLICY "delete_service_role_pedidos" ON pedidos_fornecedor FOR DELETE USING (true);
CREATE POLICY "insert_service_role_nfe" ON notas_fiscais FOR INSERT WITH CHECK (true);
CREATE POLICY "delete_service_role_nfe" ON notas_fiscais FOR DELETE USING (true);
